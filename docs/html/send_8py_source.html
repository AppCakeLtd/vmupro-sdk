227def WriteBytesChunked(inBytes, chunkSize):
228
229 bytesSent = 0
230 totalBytes = len(inBytes)
231 chunkCounter = 0
232 maxChunks = (totalBytes/chunkSize)
233 while bytesSent < totalBytes:
234
235 print(f"PC: Writing chunk {chunkCounter} / {maxChunks}")
236
237 bytesLeft = totalBytes - bytesSent
238 thisChunkSize = chunkSize
239 if (thisChunkSize > bytesLeft):
240 thisChunkSize = bytesLeft
241
242 chunk = inBytes[bytesSent: bytesSent + thisChunkSize]
243
244 if debugMode:
245 print(f"PC: Using chunk size {thisChunkSize}")
246 uart.write(chunk)
247 uart.flush()
248 bytesSent += len(chunk)
249
250 print(f"PC: Sent: {bytesSent} of {totalBytes}")
251
252 if (bytesSent < totalBytes):
253 WaitForResponse("MOREDATA", None)
254
255 chunkCounter += 1
256
257 print(f"\n\nPC: Sent {bytesSent} bytes")
